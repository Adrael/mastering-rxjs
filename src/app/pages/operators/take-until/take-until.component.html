takeUntil() emits the values emitted by the source Observable until a notifier Observable emits a value. 


      takeUntil<T>(notifier: Observable<any>): MonoTypeOperatorFunction<T>;
    

      // => Tick every second until the first click on a DIV happens

      import { fromEvent, interval } from 'rxjs';
      import { takeUntil } from 'rxjs/operators';

      const divClick$ = fromEvent(document, 'click')
      .pipe(
        filter((event: MouseEvent) => (<HTMLElement>event.target).tagName === 'DIV')
      );

      interval(1000)
      .pipe(
        takeUntil(divClick$)
      )
      .subscribe((item: number) => console.log(item));

      // Output: 0
      // Output: 1
      // Output: 2
      // Output: ...
      // Output: 4 => Click
