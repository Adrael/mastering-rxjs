concat() creates an output Observable which sequentially emits all values from given Observable and then moves on to the next. 


      concat<T, ...>(v1: ObservableInput<T>, ..., scheduler?: SchedulerLike): Observable<T | ...>;
      concat<T>(...observables: (ObservableInput<T> | SchedulerLike)[]): Observable<T>;
      concat<T, R>(...observables: (ObservableInput<any> | SchedulerLike)[]): Observable<R>;
    

      // => Concatenate a timer counting from 0 to 3 with a synchronous sequence from 1 to 10

      import { concat, interval, range } from 'rxjs';
      import { take } from 'rxjs/operators';

      const timer = interval(1000).pipe(take(4));
      const sequence = range(1, 10);
      const observable = concat(timer, sequence);

      observable.subscribe((value: number) => console.log(value));

      // Output: 0 => after 1s
      // Output: 1 => after 2s
      // Output: 2 => after 3s
      // Output: 3 => after 4s
      // Output: 1 ... 10 => immediately
