concatMap() projects each source value to an Observable which is merged in the output Observable, in a serialized fashion waiting for each one to complete before merging the next. 


      concatMap<T, R>(project: (value: T, index: number) =>  ObservableInput<R>): OperatorFunction<T, R>;
    

      // => For each click event, tick every second from 0 to 3, with no concurrency

      import { concat, interval, range } from 'rxjs';
      import { take } from 'rxjs/operators';

      import { fromEvent, interval } from 'rxjs';
      import { concatMap, take } from 'rxjs/operators';

      fromEvent(document, 'click')
      .pipe(
        concatMap((event: MouseEvent) => interval(1000).pipe(take(4)))
      )
      .subscribe((value: number) => console.log(value));

      // For every click on the "document" it will emit values 0 to 3 spaced by a 1000ms interval
      // Without any concurrency
