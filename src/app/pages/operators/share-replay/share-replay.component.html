shareReplay() shares source and replay specified number of emissions on subscription.

      import { interval } from 'rxjs';
      import { shareReplay, take } from 'rxjs/operators';

      const observable = interval(1000)
      .pipe(
        take(4),
        shareReplay(3)
      );

      observable.subscribe(x => console.log('A:', x));
      observable.subscribe(y => console.log('B:', y));

      // Output: "A: 0"
      // Output: "B: 0"
      // Output: "A: 1"
      // Output: "B: 1"
      // Output: "A: 2"
      // Output: "B: 2"
      // Output: "A: 3"
      // Output: "B: 3"
