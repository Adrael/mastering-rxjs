startWith() returns an Observable that emits the items you specify as arguments before it begins to emit items emitted by the source Observable. 


      startWith<T>(scheduler?: SchedulerLike): MonoTypeOperatorFunction<T>;
      startWith<T, D = T, ...>(v1: D, ..., scheduler?: SchedulerLike): OperatorFunction<T, T | D | ...>;
      startWith<T, D = T>(...array: Array<D | SchedulerLike>): OperatorFunction<T, T | D>;
    

      // => Start the chain of emissions with "first", "second"

      import { of } from 'rxjs';
      import { startWith } from 'rxjs/operators';

      of("from source")
      .pipe(startWith("first", "second"))
      .subscribe((item: string) => console.log(item));

      // Output: "first"
      // Output: "second"
      // Output: "from source"
