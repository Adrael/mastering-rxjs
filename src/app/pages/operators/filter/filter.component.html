filter() filters items emitted by the source Observable by only emitting those that satisfy a specified predicate. 


      filter<T, S extends T>(predicate: (value: T, index: number) => value is S, thisArg?: any): OperatorFunction<T, S>;
      filter<T>(predicate: (value: T, index: number) => boolean, thisArg?: any): MonoTypeOperatorFunction<T>;
    

      // => Emit only click events whose target was a DIV element

      import { fromEvent } from 'rxjs';
      import { filter } from 'rxjs/operators';

      fromEvent(document, 'click')
      .pipe(
        filter((event: MouseEvent) => (<HTMLElement>event.target).tagName === 'DIV')
      )
      .subscribe((event: MouseEvent) => console.log(event));
