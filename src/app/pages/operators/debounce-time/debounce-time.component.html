debounceTime() emits a value from the source Observable only after a particular time span has passed without another source emission. 


      debounceTime<T>(dueTime: number, scheduler?: SchedulerLike): MonoTypeOperatorFunction<T>
    

      // => Emit the most recent click after a burst of clicks

      import { fromEvent } from 'rxjs';
      import { debounceTime } from 'rxjs/operators';

      fromEvent(document, 'click')
      .pipe(
        debounceTime(1000)
      )
      .subscribe((event: MouseEvent) => console.log(event));

      // Output: MouseEvent  { ... }
