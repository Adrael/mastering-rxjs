combineLatest() combines multiple Observables to create an Observable whose values are calculated from the latest values of each of its input Observables. 


      combineLatest<T, ...>(v1: ObservableInput<T>, ..., scheduler?: SchedulerLike): Observable<[T, ...]>;

      combineLatest<T>(array: ObservableInput<T>[], scheduler?: SchedulerLike): Observable<T[]>;
      combineLatest<R>(array: ObservableInput<any>[], scheduler?: SchedulerLike): Observable<R>;
      combineLatest<T>(...observables: Array<ObservableInput<T> | SchedulerLike>): Observable<T[]>;
      combineLatest<T, R>(...observables: Array<ObservableInput<T> | ((...values: Array<T>) => R) | SchedulerLike>): Observable<R>;
      combineLatest<R>(...observables: Array<ObservableInput<any> | ((...values: Array<any>) => R) | SchedulerLike>): Observable<R>;
    

      // => Combine two timer Observables

      import { combineLatest, timer } from 'rxjs';

      combineLatest(
        timer(0, 1000),
        timer(500, 1000)
      )
      .subscribe((values: Array<number>) => console.log(values));

      // Output: [0, 0] => after 0.5s
      // Output: [1, 0] => after 1s
      // Output: [1, 1] => after 1.5s
      // Output: [2, 1] => after 2s
      // Output: ...
