forkJoin() combines the emissions of multiple Observables together via a specified function and emit single items for each combination based on the results of this function. 


      // forkJoin([a$, b$, c$]);
      forkJoin<T>(sources: [ObservableInput<T>]): Observable<T[]>;
      forkJoin<T, ...>(sources: [ObservableInput<T>, ...]): Observable<[T, ...]>;
      forkJoin<T>(sources: Array<ObservableInput<T>>): Observable<T[]>;
      
      // forkJoin(a$, b$, c$)
      forkJoin<T>(v1: ObservableInput<T>): Observable<T[]>;
      forkJoin<T, ...>(v1: ObservableInput<T>, ...): Observable<[T, ...]>;
    

      // => Use forkJoin with operator emitting after some time

      import { forkJoin, interval } from 'rxjs';
      import { take } from 'rxjs/operators';

      forkJoin(
        interval(1000).pipe(take(3)), // emit 0, 1, 2 every second and complete
        interval(500).pipe(take(4)),  // emit 0, 1, 2, 3 every half a second and complete
      )
      .subscribe((values: Array<number>) => console.log(values));

      // Output: [2, 3] => after 3 seconds
