last() returns an Observable that emits only the last item emitted by the source Observable. It optionally takes a predicate function as a parameter, in which case, rather than emitting the last item from the source Observable, the resulting Observable will emit the last item from the source Observable that satisfies the predicate. 


      last<T, D = T>(predicate?: null, defaultValue?: D): OperatorFunction<T, T | D>;
      last<T, S extends T>(predicate: (value: T, index: number, source: Observable<T>) => value is S, defaultValue?: S): OperatorFunction<T, S>;
      last<T, D = T>(predicate: (value: T, index: number, source: Observable<T>) => boolean, defaultValue?: D): OperatorFunction<T, T | D>;
    

      // => Emits the last value matching given predicate function

      import { from } from 'rxjs';
      import { last } from 'rxjs/operators';

      from([1, 2, 3, 4, 5])
      .pipe(
        last((value: number) => value % 2 === 0)
      )
      .subscribe((value: number) => console.log(value));

      // Output: 4
