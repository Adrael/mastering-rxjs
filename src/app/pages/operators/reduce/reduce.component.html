reduce() applies an accumulator function over the source Observable, and returns the accumulated result when the source completes, given an optional seed value. 


      reduce<T>(accumulator: (acc: T, value: T, index: number) => T, seed?: T): MonoTypeOperatorFunction<T>;
      reduce<T>(accumulator: (acc: T[], value: T, index: number) => T[], seed: T[]): OperatorFunction<T, T[]>;
      reduce<T, R>(accumulator: (acc: R, value: T, index: number) => R, seed?: R): OperatorFunction<T, R>;
    

      // => Count the number of click events that happened in 5 seconds

      import { fromEvent, interval }  from 'rxjs';
      import { mapTo, reduce, takeUntil }  from 'rxjs/operators';

      fromEvent(document, 'click')
      .pipe(
        takeUntil(interval(5000)),
        mapTo(1),
        reduce((accumulated: number, one: number) => accumulated + one, 0)
      );
      .subscribe((value: number) => console.log(value));

      // Output: ... => Number of click during last 5 seconds
