scan() applies an accumulator function over the source Observable, and returns each intermediate result, with an optional seed value. 


      scan<T>(accumulator: (acc: T, value: T, index: number) => T, seed?: T): MonoTypeOperatorFunction<T>;
      scan<T>(accumulator: (acc: T[], value: T, index: number) => T[], seed?: T[]): OperatorFunction<T, T[]>;
      scan<T, R>(accumulator: (acc: R, value: T, index: number) => R, seed?: R): OperatorFunction<T, R>;
    

      // => Count the number of click events

      import { fromEvent } from 'rxjs';
      import { mapTo, scan } from 'rxjs/operators';

      fromEvent(document, 'click')
      .pipe(
        mapTo(1),
        scan((accumulated: number, one: number) => accumulated + one, 0)
      )
      .subscribe((value: number) => console.log(value));

      // Output: 1
      // Output: 2
      // Output: 3
