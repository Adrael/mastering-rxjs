The ReplaySubject "replays" or emits old values to new subscribers. It buffers a set number of values and will emit those values immediately to any new subscribers in addition to emitting new values to existing subscribers. 


      ReplaySubject<T>(bufferSize?: number, windowTime?: number, scheduler?: SchedulerLike): Subject<T>;
    

      import { ReplaySubject } from 'rxjs';

      const replaySubject = new ReplaySubject<number>(2);
      replaySubject.next(0);
      replaySubject.next(1);
      replaySubject.subscribe((value: number) => console.log('First:', value));
      replaySubject.next(2);
      replaySubject.subscribe((value: number) => console.log('Second:', value));
      replaySubject.next(3);
      replaySubject.subscribe((value: number) => console.log('Third:', value));

      // Output: First: 0
      // Output: First: 1
      // Output: First: 2
      // Output: Second: 1
      // Output: Second: 2
      // Output: First: 3
      // Output: Second: 3
      // Output: Third: 2
      // Output: Third: 3
