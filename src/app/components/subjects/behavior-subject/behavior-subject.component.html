BehaviorSubject<T>(initialValue: T): Subject<T>;
    

      import { BehaviorSubject } from 'rxjs';

      const behaviorSubject = new BehaviorSubject<number>(null);
      behaviorSubject.subscribe((value: number) => console.log(value));
      behaviorSubject.next(0);
      behaviorSubject.next(1);
      behaviorSubject.next(2);
      behaviorSubject.next(3);

      // Output: null
      // Output: 0
      // Output: 1
      // Output: 2
      // Output: 3

A BehaviorSubject requires an initial value and emits its current value whenever it is subscribed to.
